73  ResultType * resultCasted = mtsGenericTypes<ResultType>::CastArg(result);

82  template <typename _dummy>
83  class ConditionalCast<true, _dummy> {
84  // Specialization: ReturnType is derived from mtsGenericObjectProxy (and thus also from mtsGenericObject)
85  // In this case, we may need to create a temporary Proxy object.
86  public:
87  static mtsExecutionResult CallMethod(ClassType * classInstantiation, ActionType action, mtsGenericObject & result) {
88  // First, check if a Proxy object was passed.
89  ResultType * resultCasted = dynamic_cast<ResultType *>(&result);
90  if (resultCasted) {
91  (classInstantiation->*action)(*resultCasted);
92  return mtsExecutionResult::COMMAND_SUCCEEDED;
93  }
94  // If it isn't a Proxy, maybe it is a ProxyRef
95  typedef typename ResultType::RefType ResultRefType;
96  ResultRefType * dataRef = dynamic_cast<ResultRefType *>(&result);
97  if (!dataRef) {
98  CMN_LOG_INIT_ERROR << "mtsCallableVoidReturn: CallMethod could not cast from " << typeid(result).name()
99  << " to " << typeid(ResultRefType).name() << std::endl;
100  return mtsExecutionResult::INVALID_INPUT_TYPE;
101  }
102  // Now, make the call using the temporary
103  ResultType temp;
104  (classInstantiation->*action)(temp);
105  // Finally, copy the data to the return
106  *dataRef = temp;
107  return mtsExecutionResult::COMMAND_SUCCEEDED;
108  }
109  };
