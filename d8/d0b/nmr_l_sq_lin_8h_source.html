171  static inline CISSTNETLIB_INTEGER GetWorkspaceSize(CISSTNETLIB_INTEGER ma, CISSTNETLIB_INTEGER me, CISSTNETLIB_INTEGER mg, CISSTNETLIB_INTEGER n)
172  {
173  CISSTNETLIB_INTEGER minmn = -1;
174  CISSTNETLIB_INTEGER lwork = -1;
175  CISSTNETLIB_INTEGER k = -1;
176  if ((me == 0) && (mg ==0)) {// case LS
177  minmn = (ma < n)?ma:n;
178  minmn = (1 > minmn)?1:minmn;
179  lwork = 2*minmn;
180  return lwork;
181  } else if (me == 0) { // case LSI
182  k = (ma+mg>n)?(ma+mg):n;
183  return k+n+(mg+2)*(n+7);
184  } else { // case LSEI
185  k = (ma+mg>n)?(ma+mg):n;
186  return 2*(me+n)+k+(mg+2)*(n+7);
187  }
188  }

190  {
191  if ((me == 0) && (mg ==0)) {// case LS
192  return 0;
193  } else if (me == 0) { // case LSI
194  return mg+2*n+1;
195  } else { // case LSEI
196  return mg+2*n+2;
197  }
