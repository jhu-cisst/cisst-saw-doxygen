401  ConstVectorPointerType & RowPointers(ConstVectorPointerType & rowPointers) const throw(std::runtime_error) {
402  if (! this->col_stride() == 1) {
403  cmnThrow(std::runtime_error("vctDynamicMatrix: RowPointers requires compact rows"));
404  }
405  const size_type rows = this->rows();
406  // resize the vector
407  rowPointers.SetSize(rows);
408  index_type index;
409  for (index = 0; index < rows; ++index) {
410  rowPointers[index] = this->Row(index).Pointer();
411  }
412  return rowPointers;
413  }
