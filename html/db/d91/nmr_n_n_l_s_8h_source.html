805 template <class _matrixOwnerType>
806 inline CISSTNETLIB_INTEGER nmrNNLS(vctDynamicMatrixBase<_matrixOwnerType, CISSTNETLIB_DOUBLE> & A,
807  nmrNNLSDynamicData & data)
808  throw (std::runtime_error)
809 {
810  typename nmrNNLSDynamicData::Friend dataFriend(data);
811  CISSTNETLIB_INTEGER Info;
812  char m_Jobu = 'A';
813  char m_Jobvt = 'A';
814  CISSTNETLIB_INTEGER m_Lwork = static_cast<CISSTNETLIB_INTEGER>(nmrNNLSDynamicData::WorkspaceSize(dataFriend.M(),
815  dataFriend.N()));
816  /* check that storage order matches with Allocate() */
817  if (A.StorageOrder() != dataFriend.StorageOrder()) {
818  cmnThrow(std::runtime_error("nmrNNLS: Storage order used for Allocate was different"));
819  }
820  /* check sizes */
821  if ((dataFriend.M() != A.rows()) || (dataFriend.N() != A.cols())) {
822  cmnThrow(std::runtime_error("nmrNNLS: Size used for Allocate was different"));
823  }
824  /* check that the matrices are compact */
825  if (! A.IsCompact()) {
826  cmnThrow(std::runtime_error("nmrNNLS: Requires a compact matrix"));
827  }
828 
829  /* Based on storage order, permute U and Vt as well as dimension */
830  CISSTNETLIB_DOUBLE *UPtr, *VtPtr;
831  CISSTNETLIB_INTEGER m_Lda, m_Ldu, m_Ldvt;
832 
833  if (A.IsColMajor()) {
834  m_Lda = (1 > dataFriend.M()) ? 1 : dataFriend.M();
835  m_Ldu = dataFriend.M();
836  m_Ldvt = dataFriend.N();
837  UPtr = dataFriend.U().Pointer();
838  VtPtr = dataFriend.Vt().Pointer();
839  } else if (A.IsRowMajor()) {
840  m_Lda = (1 > dataFriend.N()) ? 1 : dataFriend.N();
841  m_Ldu = dataFriend.N();
842  m_Ldvt = dataFriend.M();
843  UPtr = dataFriend.Vt().Pointer();
844  VtPtr = dataFriend.U().Pointer();
845  }
846 
847  // for versions based on gfortran/lapack, CISSTNETLIB_VERSION is
848  // defined
849 #if defined(CISSTNETLIB_VERSION)
850 #if defined(CISSTNETLIB_VERSION_MAJOR)
851 #if (CISSTNETLIB_VERSION_MAJOR >= 3)
852  cisstNetlib_dgesvd_(&m_Jobu, &m_Jobvt, &m_Ldu, &m_Ldvt,
853  A.Pointer(), &m_Lda, dataFriend.S().Pointer(),
854  UPtr, &m_Ldu,
855  VtPtr, &m_Ldvt,
856  dataFriend.Workspace().Pointer(), &m_Lwork, &Info);
857 #endif
858 #else // no major version
859  dgesvd_(&m_Jobu, &m_Jobvt, &m_Ldu, &m_Ldvt,
860  A.Pointer(), &m_Lda, dataFriend.S().Pointer(),
861  UPtr, &m_Ldu,
862  VtPtr, &m_Ldvt,
863  dataFriend.Workspace().Pointer(), &m_Lwork, &Info);
864 #endif // CISSTNETLIB_VERSION
865 #else
866  ftnlen jobu_len = (ftnlen)1, jobvt_len = (ftnlen)1;
867  la_dzlapack_MP_sgesvd_nat(&m_Jobu, &m_Jobvt, &m_Ldu, &m_Ldvt,
868  A.Pointer(), &m_Lda, dataFriend.S().Pointer(),
869  UPtr, &m_Ldu,
870  VtPtr, &m_Ldvt,
871  dataFriend.Workspace().Pointer(), &m_Lwork, &Info,
872  jobu_len, jobvt_len);
873 #endif
874 
875  return Info;
876 }
