945  void ToStream(std::ostream & outputStream) const
946  {
947  size_type index;
948  const size_type mySize = size();
949  // preserve the formatting flags as they were
950  const std::streamsize width = outputStream.width(12);
951  const std::streamsize precision = outputStream.precision(6);
952  bool showpoint = ((outputStream.flags() & std::ios_base::showpoint) != 0);
953  outputStream << std::setprecision(6) << std::showpoint;
954  for (index = 0; index < mySize; ++index) {
955  outputStream << std::setw(12) << (*this)[index];
956  if (index < (mySize-1)) {
957  outputStream << " ";
958  }
959  }
960  // resume the formatting flags
961  outputStream << std::setprecision(precision) << std::setw(width);
962  if (!showpoint) {
963  outputStream << std::noshowpoint;
964  }
965  }
