361  {
362  // get and set size

365  cmnDeSerializeRaw(inputStream, myRows);

367 
368  if ((myRows != this->rows()) || (myCols != this->cols())) {
369  cmnThrow(std::runtime_error("vctDynamicMatrixRef::DeSerializeRaw: Sizes of matrices don't match"));

371 
372  // get data
373  size_type indexRow, indexCol;
374  for (indexRow = 0; indexRow < myRows; ++indexRow) {
