783 template <class _matrixOwnerType>
784 inline CISSTNETLIB_INTEGER nmrPInverse(vctDynamicMatrixBase<_matrixOwnerType, CISSTNETLIB_DOUBLE> &A, nmrPInverseDynamicData &data) throw (std::runtime_error)
785 {
786  typedef vct::size_type size_type;
787 
788  typename nmrPInverseDynamicData::Friend dataFriend(data);
789  CISSTNETLIB_INTEGER ret_value;
790  /* check that the size and storage order matches with Allocate() */
791  if (A.StorageOrder() != dataFriend.StorageOrder()) {
792  cmnThrow(std::runtime_error("nmrPInverse Solve: Storage order used for Allocate was different"));
793  }
794  if ((A.rows() != dataFriend.M()) || (A.cols() != dataFriend.N())) {
795  cmnThrow(std::runtime_error("nmrPInverse Solve: Size used for Allocate was different"));
796  }
797  const size_type rows = A.rows();
798  const size_type cols = A.cols();
799  const size_type minmn = (rows < cols) ? rows : cols;
800  const size_type maxmn = (rows > cols) ? rows : cols;
801 
802  ret_value = nmrSVD(A, dataFriend.U(), dataFriend.S(),
803  dataFriend.Vt(), dataFriend.Workspace());
804  const CISSTNETLIB_DOUBLE eps = cmnTypeTraits<CISSTNETLIB_DOUBLE>::Tolerance() * dataFriend.S().at(0) * maxmn;
805 
806  dataFriend.PInverse().SetAll(0);
807  CISSTNETLIB_DOUBLE singularValue;
808  size_type irank, i, j;
809  for (irank = 0; irank < minmn; irank++) {
810  if ((singularValue = dataFriend.S().at(irank)) > eps) {
811  for (j = 0; j < rows; j++) {
812  for (i = 0; i < cols; i++) {
813  dataFriend.PInverse().at(i, j) = dataFriend.PInverse().at(i, j)
814  + dataFriend.Vt().at(irank, i) * dataFriend.U().at(j, irank) / singularValue;
815  }
816  }
817  }
818  }
819  return ret_value;
820 }

890 template <vct::size_type _rows, vct::size_type _cols, vct::size_type _work, bool _storageOrder, class _dataPtrType>
891 inline CISSTNETLIB_INTEGER nmrPInverse(vctFixedSizeMatrix<CISSTNETLIB_DOUBLE, _rows, _cols, _storageOrder> & A,
892  vctFixedSizeMatrix<CISSTNETLIB_DOUBLE, _cols, _rows, _storageOrder> & pInverse,
893  vctFixedSizeVectorBase<_work, 1, CISSTNETLIB_DOUBLE, _dataPtrType> & workspace)
894 {

896  const size_type lwork = nmrPInverseFixedSizeData<_rows, _cols, _storageOrder>::LWORK;
897  const size_type lwork_3 = nmrPInverseFixedSizeData<_rows, _cols, _storageOrder>::LWORK_3;
898  const size_type minmn = nmrPInverseFixedSizeData<_rows, _cols, _storageOrder>::MIN_MN;
899  const size_type maxmn = (_rows > _cols) ? _rows : _cols;
900  //Assert if requirement is greater than size provided!
901  CMN_ASSERT(lwork <= _work);
902  // split Work into submatrices and vectors
903  // for this we simply use the nmrSVD with dynamic ref as parameters as we dont have
904  // nmrSVD with fixedsizeref as parameters
905  vctDynamicMatrixRef<CISSTNETLIB_DOUBLE> ARef(A);
906  vctDynamicMatrixRef<CISSTNETLIB_DOUBLE> URef(_rows, _rows,
907  (_storageOrder) ? _rows : 1,
908  (_storageOrder) ? 1 : _rows,
909  workspace.Pointer(0));
910  vctDynamicMatrixRef<CISSTNETLIB_DOUBLE> VtRef(_cols, _cols,
911  (_storageOrder) ? _cols : 1,
912  (_storageOrder) ? 1 : _cols,
913  workspace.Pointer(_rows*_rows));
914  vctDynamicVectorRef<CISSTNETLIB_DOUBLE> SRef(minmn,
915  workspace.Pointer(_rows * _rows + _cols * _cols));
916  vctDynamicVectorRef<CISSTNETLIB_DOUBLE> SVDWorkspaceRef(lwork_3,
917  workspace.Pointer(_rows * _rows + _cols * _cols + minmn));
918  vctDynamicMatrixRef<CISSTNETLIB_DOUBLE> PRef(_cols, _rows, (_storageOrder) ? _rows : 1, (_storageOrder) ? 1 : _cols,
919  workspace.Pointer(_rows * _rows + _cols * _cols + minmn + lwork_3));
920  vctDynamicMatrixRef<CISSTNETLIB_DOUBLE> pInverseRef(pInverse);
921  CISSTNETLIB_INTEGER ret_value;
922  ret_value = nmrSVD(ARef, URef, SRef, VtRef, SVDWorkspaceRef);
923  const CISSTNETLIB_DOUBLE eps = cmnTypeTraits<CISSTNETLIB_DOUBLE>::Tolerance() * SRef(0) * maxmn;
924  /*
925  vctDynamicMatrix<CISSTNETLIB_DOUBLE> SM (_cols, _rows, _storageOrder);
926  SM.SetAll(0.);
927  for (size_type irank = 0; irank < minmn; irank++) {
928  if (SRef(irank) > eps) {
929  SM(irank, irank) = 1.0/SRef(irank);
930  }
931  }
932  PRef.ProductOf(SM, URef.Transpose());
933  */
934  size_type irank;
935  PRef.SetAll(0.);
936  for (irank = 0; irank < minmn; irank++) {
937  if (SRef(irank) > eps) {
938  PRef.Row(irank).ProductOf(URef.Transpose().Row(irank), 1.0/SRef(irank));
939  } else {
940  PRef.Row(irank).SetAll(0.);
941  }
942  }
943  pInverseRef.ProductOf(VtRef.Transpose(), PRef);
944 
945  return ret_value;
946 }
