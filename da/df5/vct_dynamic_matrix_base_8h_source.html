449  inline bool Zeros(void) {
450  if (this->IsCompact()) {
451  memset(this->Pointer(), 0, this->size() * sizeof(value_type));
452  return true;
453  } else if (this->col_stride() == 1) {
454  /* memset row by row */
455  const size_type sizeOfRow = this->cols() * sizeof(value_type);
456  const stride_type rowStride = this->row_stride();
457  pointer currentPointer = this->Pointer();
458  const pointer endPointer = currentPointer + this->rows() * rowStride;
459  for (;
460  currentPointer != endPointer;
461  currentPointer += rowStride) {
462  memset(currentPointer, 0, sizeOfRow);
463  }
464  return true;
465  } else if (this->row_stride() == 1) {
466  /* memset col by col */
467  const size_type sizeOfCol = this->rows() * sizeof(value_type);
468  const stride_type colStride = this->col_stride();
469  pointer currentPointer = this->Pointer();
470  const pointer endPointer = currentPointer + this->cols() * colStride;
471  for (;
472  currentPointer != endPointer;
473  currentPointer += colStride) {
474  memset(currentPointer, 0, sizeOfCol);
475  }
476  return true;
477  } else {
478  this->SetAll(static_cast<value_type>(0));
479  return false;
480  }
481  }
