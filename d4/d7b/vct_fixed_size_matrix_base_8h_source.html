424  inline bool Zeros(void) {
425  if (this->IsCompact()) {
426  memset(this->Pointer(), 0, this->size() * sizeof(value_type));
427  return true;
428  } else if (this->col_stride() == 1) {
429  /* memset row by row */
430  const size_type sizeOfRow = this->cols() * sizeof(value_type);
431  const stride_type rowStride = this->row_stride();
432  pointer currentPointer = this->Pointer();
433  const pointer endPointer = currentPointer + this->rows() * rowStride;
434  for (;
435  currentPointer != endPointer;
436  currentPointer += rowStride) {
437  memset(currentPointer, 0, sizeOfRow);
438  }
439  return true;
440  } else if (this->row_stride() == 1) {
441  /* memset col by col */
442  const size_type sizeOfCol = this->rows() * sizeof(value_type);
443  const stride_type colStride = this->col_stride();
444  pointer currentPointer = this->Pointer();
445  const pointer endPointer = currentPointer + this->cols() * colStride;
446  for (;
447  currentPointer != endPointer;
448  currentPointer += colStride) {
449  memset(currentPointer, 0, sizeOfCol);
450  }
451  return true;
452  } else {
453  this->SetAll(static_cast<value_type>(0));
454  return false;
455  }
456  }
